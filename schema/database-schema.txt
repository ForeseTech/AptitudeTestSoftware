CREATE DATABASE MOCKS;
USE MOCKS;

CREATE TABLE QUESTIONS (
	QID INT(10) AUTO_INCREMENT PRIMARY KEY,
	QUESTION VARCHAR(500) NOT NULL UNIQUE,
	OPTIONA VARCHAR(80) NOT NULL,
	OPTIONB VARCHAR(80) NOT NULL,
	OPTIONC VARCHAR(80) NOT NULL,
	OPTIOND VARCHAR(80) NOT NULL,
	ANSWER ENUM("A","B","C","D") NOT NULL,
	DEPARTMENT ENUM("ALL", "CSE", "EEE", "ECE", "BIO", "MEC", "AUT", "CIV", "CHE", "IT") NOT NULL DEFAULT "ALL",
	QSET ENUM("1","2","3") NOT NULL,
	QIMAGE VARCHAR(30) NOT NULL DEFAULT "NONE"
);

CREATE TABLE USERS (
	TERMINAL INT(10) PRIMARY KEY AUTO_INCREMENT,
	LOGGEDIN INT(1) NOT NULL DEFAULT 0,
	NAME VARCHAR(200) NOT NULL DEFAULT "None",
	REGNUMBER VARCHAR(10) NOT NULL DEFAULT "000000000",
	DEPARTMENT ENUM("NULL", "CSE", "EEE", "ECE", "BIO", "MEC", "AUT", "CIV", "CHE", "IT") NOT NULL DEFAULT "NULL",
	EMAIL VARCHAR(100) NOT NULL DEFAULT "example@example.com"
);

CREATE TABLE RESULTS (
	REGNUMBER VARCHAR(10) PRIMARY KEY,
	SECTION_ONE INT(5) NOT NULL DEFAULT 0,
	SECTION_TWO INT(5) NOT NULL DEFAULT 0,
	SECTION_THREE INT(5) NOT NULL DEFAULT 0,
	SECTION_FOUR INT(5) NOT NULL DEFAULT 0,
	SECTION_FIVE INT(5) NOT NULL DEFAULT 0,
	SECTION_TECH INT(5) NOT NULL DEFAULT 0,
	TOTAL INT(5) NOT NULL DEFAULT 0
);
